183    void updateItemMap(Widget widget, boolean toAdd) {
184        // Option ==> update with this option
185        if (widget instanceof Option) {
186            Option option = (Option) widget;
187            if (toAdd)
188                itemMap.put(option.getSelectElement(), option);
189            else
190                itemMap.remove(option.getSelectElement());
191        } else if (widget instanceof OptGroup) {
192            // OptGroup ==> update with all optGroup options
193            OptGroup optGroup = (OptGroup) widget;
194            if (toAdd)
195                itemMap.putAll(optGroup.getItemMap());
196            else
197                for (Entry<OptionElement, Option> entry : optGroup.getItemMap().entrySet()) {
198                    OptionElement optElem = entry.getKey();
199                    itemMap.remove(optElem);
200                }
201        }
202    }

530    private void updateStyle() {
531        StringBuilder sb = new StringBuilder();
532        if (type != null) {
533            sb.append(type.getCssName());
534        }
535        if (size != null) {
536            if (!sb.toString().isEmpty()) {
537                sb.append(" ");
538            }
539            sb.append(size.getCssName());
540        }
541        setStyle(sb.toString());
542    }

635    @Override
636    public void setValue(final T value, final boolean fireEvents) {
637
638        T oldValue = fireEvents ? getValue() : null;
639
640        setSelectedValue(value);
641
642        if (fireEvents) {
643            T newValue = getValue();
644            ValueChangeEvent.fireIfNotEqual(this, oldValue, newValue);
645        }
646    }

721        List<Option> selectedItems = new ArrayList<>(0);
722        NodeList<OptionElement> items = selectElement.getOptions();
723        for (int i = 0; i < items.getLength(); i++) {
724            OptionElement item = items.getItem(i);
725            Option option = itemMap.get(item);
726            if (option != null)
727                selectedItems.add(option);
728        }
729        return selectedItems;
730    }

732    /**
733     * Returns <code>true</code> if the item at the given index is selected.<br>
734     * <br>
735     * <b>Note</b>: if the item at the given index is a divider, this method
736     * always returns <code>false</code>.
737     *
738     * @param index
739     * @return
740     */
741    public boolean isItemSelected(final int index) {
742        checkIndex(index);
743        OptionElement item = selectElement.getOptions().getItem(index);
744        Option option = itemMap.get(item);
745        return option != null && option.isSelected();
746    }

748    /**
749     * Returns the {@link Option} at the given index.
750     *
751     * @param index
752     * @return
753     */
754    public Option getItem(final int index) {
755        checkIndex(index);
756        OptionElement item = selectElement.getOptions().getItem(index);
757        return itemMap.get(item);
758    }

760    private void checkIndex(final int index) {
761        int max = getItemCount();
762        if (index < 0 || index >= max) {
763            throw new IndexOutOfBoundsException("Index should be in [0, " + max + "]");
764        }
765    }
